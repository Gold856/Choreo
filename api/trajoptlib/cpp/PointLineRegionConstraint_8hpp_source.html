50 void Apply(sleipnir::OptimizationProblem& problem, const Pose2v& pose,
51 [[maybe_unused]] const Translation2v& linearVelocity,
52 [[maybe_unused]] const sleipnir::Variable& angularVelocity,
53 [[maybe_unused]] const Translation2v& linearAcceleration,
54 [[maybe_unused]] const sleipnir::Variable& angularAcceleration) {
55 auto bumperCorner =
56 pose.Translation() + m_robotPoint.RotateBy(pose.Rotation());
57
58 // Rearrange y − y₀ = m(x − x₀) where m = (y₁ − y₀)/(x₁ − x₀) into ax + by =
59 // c form.
60
61 // y − y₀ = m(x − x₀)
62 // y − y₀ = (y₁ − y₀)/(x₁ − x₀)(x − x₀)
63 // (x₁ − x₀)(y − y₀) = (y₁ − y₀)(x − x₀)
64 // (x₁ − x₀)y − (x₁ − x₀)y₀ = (y₁ − y₀)x − (y₁ − y₀)x₀
65 // (y₀ − y₁)x + (x₁ − x₀)y = −(y₁ − y₀)x₀ + (x₁ − x₀)y₀
66 // (y₀ − y₁)x + (x₁ − x₀)y = (y₀ − y₁)x₀ + (x₁ − x₀)y₀
67
68 // ax + by = c where
69 // a = y₀ − y₁
70 // b = x₁ − x₀
71 // c = (y₀ − y₁)x₀ + (x₁ − x₀)y₀
72 // = ax₀ + by₀
73
74 // ax + by = ax₀ + by₀ where
75 // a = y₀ − y₁
76 // b = x₁ − x₀
77
78 auto a = m_fieldLineStart.Y() - m_fieldLineEnd.Y();
79 auto b = m_fieldLineEnd.X() - m_fieldLineStart.X();
80
81 problem.SubjectTo(a * bumperCorner.X() + b * bumperCorner.Y() >
82 a * m_fieldLineStart.X() + b * m_fieldLineStart.Y());
83 }
