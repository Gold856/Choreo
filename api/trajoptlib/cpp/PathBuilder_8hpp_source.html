132 void WptObstacle(size_t index, const Obstacle& obstacle) {
133 for (auto& _bumpers : bumpers) {
134 auto minDistance = _bumpers.safetyDistance + obstacle.safetyDistance;
135
136 size_t bumperCornerCount = _bumpers.points.size();
137 size_t obstacleCornerCount = obstacle.points.size();
138 if (bumperCornerCount == 1 && obstacleCornerCount == 1) {
139 // if the bumpers and obstacle are only one point
140 WptConstraint(
141 index, PointPointMinConstraint{_bumpers.points.at(0),
142 obstacle.points.at(0), minDistance});
143 return;
144 }
145
146 // robot bumper edge to obstacle point constraints
147 for (auto& obstaclePoint : obstacle.points) {
148 // First apply constraint for all but last edge
149 for (size_t bumperCornerIndex = 0;
150 bumperCornerIndex < bumperCornerCount - 1; ++bumperCornerIndex) {
151 WptConstraint(index, LinePointConstraint{
152 _bumpers.points.at(bumperCornerIndex),
153 _bumpers.points.at(bumperCornerIndex + 1),
154 obstaclePoint, minDistance});
155 }
156 // apply to last edge: the edge connecting the last point to the first
157 // must have at least three points to need this
158 if (bumperCornerCount >= 3) {
159 WptConstraint(index,
160 LinePointConstraint{
161 _bumpers.points.at(bumperCornerCount - 1),
162 _bumpers.points.at(0), obstaclePoint, minDistance});
163 }
164 }
165
166 // obstacle edge to bumper corner constraints
167 for (auto& bumperCorner : _bumpers.points) {
168 if (obstacleCornerCount > 1) {
169 for (size_t obstacleCornerIndex = 0;
170 obstacleCornerIndex < obstacleCornerCount - 1;
171 ++obstacleCornerIndex) {
172 WptConstraint(
173 index,
174 PointLineConstraint{
175 bumperCorner, obstacle.points.at(obstacleCornerIndex),
176 obstacle.points.at(obstacleCornerIndex + 1), minDistance});
177 }
178 if (obstacleCornerCount >= 3) {
179 WptConstraint(index, PointLineConstraint{
180 bumperCorner,
181 obstacle.points.at(bumperCornerCount - 1),
182 obstacle.points.at(0), minDistance});
183 }
184 } else {
185 WptConstraint(index,
186 PointPointMinConstraint{
187 bumperCorner, obstacle.points.at(0), minDistance});
188 }
189 }
190 }
191 }

285 void SgmtConstraint(size_t fromIndex, size_t toIndex,
286 const Constraint& constraint) {
287 assert(fromIndex < toIndex);
288
289 NewWpts(toIndex);
290 path.waypoints.at(fromIndex).waypointConstraints.push_back(constraint);
291 for (size_t index = fromIndex + 1; index <= toIndex; ++index) {
292 path.waypoints.at(index).waypointConstraints.push_back(constraint);
293 path.waypoints.at(index).segmentConstraints.push_back(constraint);
294 }

361 void NewWpts(size_t finalIndex) {
362 if (finalIndex < path.waypoints.size()) {
363 return;
364 }
365
366 for (size_t i = path.waypoints.size(); i <= finalIndex; ++i) {
367 path.waypoints.emplace_back();
368 initialGuessPoints.emplace_back(std::vector<Pose2d>{{0.0, 0.0, {0.0}}});
369 if (i != 0) {
370 controlIntervalCounts.push_back(40);
371 }
372 }
